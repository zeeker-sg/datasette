    ```python
    # ============================================================
    # Phase 6 — FTS discovery + execution helpers
    # ============================================================

    async def discover_searchable_tables(client: httpx.AsyncClient) -> dict[str, list[str]]:
        """Return {db_name: [table_names_with_fts]}.

        Live-verified: each /{db}.json table dict carries fts_table (string-or-None).
        Boot tolerance: any httpx error → empty dict; routes_search renders 503 friendly
        when cache empty and q non-empty (RESEARCH Pitfall 10).
        """
        out: dict[str, list[str]] = {}
        try:
            dbs = await fetch_databases(client)
        except httpx.HTTPError:
            return out
        for entry in dbs:
            db = entry.get("name")
            if not db:
                continue
            try:
                payload = await fetch_database(client, db)
            except httpx.HTTPError:
                continue
            if payload is None:
                continue
            names: list[str] = []
            for t in payload.get("tables") or []:
                if t.get("hidden"):
                    continue
                if t.get("name", "").startswith("_zeeker"):
                    continue
                if t.get("fts_table"):
                    names.append(t["name"])
            if names:
                out[db] = names
        return out


    async def search_table(
        client: httpx.AsyncClient,
        db: str,
        table: str,
        q: str,
        size: int = 10,
    ) -> dict | None:
        """GET /{db}/{table}.json?_search=q&_size=size&_shape=objects. None on 404."""
        r = await client.get(
            f"/{db}/{table}.json",
            params={"_search": q, "_size": size, "_shape": "objects"},
        )
        if r.status_code == 404:
            return None
        r.raise_for_status()
        return r.json()


    async def execute_sql(
        client: httpx.AsyncClient,
        db: str,
        sql: str,
        params: dict[str, Any] | None = None,
    ) -> tuple[dict | None, str | None]:
        """Execute read-only SQL against datasette /{db}.json?sql=…&_param_<name>=…

        Returns (body, error):
          - (body, None)              on 200 success
          - (body, body.error)        on 200 with body.error populated (rare)
          - (None, body.error or 'Query failed')  on 400 (RESEARCH Pitfall 1)
          - (None, 'Database not found')          on 404
          - raises httpx.HTTPError                on 5xx

        Always sends _shape=objects. Binds params via _param_<name>; NEVER concats
        param values into the SQL string.
        """
        ds_params: dict[str, Any] = {"sql": sql, "_shape": "objects"}
        for name, value in (params or {}).items():
            ds_params[f"_param_{name}"] = value
        r = await client.get(f"/{db}.json", params=ds_params)
        if r.status_code == 404:
            return None, "Database not found"
        body = r.json()
        if r.status_code == 400:
            return None, body.get("error") or "Query failed"
        r.raise_for_status()
        return body, body.get("error")
    ```

    Critical: 404 check happens BEFORE `r.json()` and BEFORE `raise_for_status()` —
    matches `fetch_database:36-39` idiom. 400 check happens BEFORE `raise_for_status()`

    """Changelog loader — reads data/changelog.yaml at lifespan boot.

    Phase 6 (D-12). Replaces M1 plugins/strings.yaml recent_updates block.
    """
    from __future__ import annotations

    from pathlib import Path

    def load_changelog() -> list[dict]:
        """Return list of {date, type, title, description} dicts.
